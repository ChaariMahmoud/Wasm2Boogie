29  private void ResolveStateVariables()
30  {
31  ResolutionHelper resolutionHelper = new ResolutionHelper(context);
32  List<ContractDefinition> sortedContracts = resolutionHelper.TopologicalSortByDependency(context.ContractDefinitions);
33 
34  foreach (ContractDefinition contract in sortedContracts)
35  {
36  // create a deep copy
37  List<int> linearizedBaseContractIds = new List<int>(contract.LinearizedBaseContracts);
38  linearizedBaseContractIds.Reverse();
39 
40  foreach (int id in linearizedBaseContractIds)
41  {
42  ContractDefinition baseContract = context.GetASTNodeById(id) as ContractDefinition;
43  Debug.Assert(baseContract != null);
44 
45  if (baseContract == contract)
46  {
47  HashSet<VariableDeclaration> stateVars = context.GetStateVarsByContract(contract);
48  foreach (VariableDeclaration stateVar in stateVars)
49  {
50  string name = stateVar.Name;
51  context.AddStateVarToDynamicType(name, contract, stateVar);
52  }
53  }
54  else
55  {
56  HashSet<VariableDeclaration> stateVars = context.GetStateVarsByContract(baseContract);
57  foreach (VariableDeclaration stateVar in stateVars)
58  {
59  if (stateVar.Visibility == EnumVisibility.PRIVATE) continue;
60 
61  string name = stateVar.Name;
62  context.AddStateVarToDynamicType(name, contract, stateVar);
63  }
64  }
65  }
66  }
