206  public override bool Visit(FunctionCall node)
207  {
208  //a function call may be of the form
209  // foo(x) | foo.value(y)(x) | foo.gas(z)(x) | foo.value(z).gas(y)(x) | foo.gas(x).value(y)(z)
210  // e.foo(x) | e.foo.value(y)(x) | e.foo.gas(z)(x) | e.foo.value(z).gas(y)(x) | e.foo.gas(x).value(y)(z)
211  // foo could be "call" as well
212  //
213  //
214  // let us remove value/gas attributes and remember then
215 
216 
217  node.MsgGas = null;
218  node.MsgValue = null;
219 
220  if (node.Expression is FunctionCall functionCall)
221  {
222  if (functionCall.Expression is MemberAccess ma)
223  {
224  // first level
225  if (ma.MemberName.Equals("value") ||
226  ma.MemberName.Equals("gas"))
227  {
228  node.Expression = ma.Expression;
229  node.MsgValue = ma.MemberName.Equals("value") ? functionCall.Arguments[0] : node.MsgValue;
230  node.MsgGas = ma.MemberName.Equals("gas") ? functionCall.Arguments[0]: node.MsgGas;
231 
232  // second level
233  if (ma.Expression is FunctionCall nestedFunctionCall)
234  {
235  if (nestedFunctionCall.Expression is MemberAccess nestedMa)
236  {
237  if (nestedMa.MemberName.Equals("value") ||
238  nestedMa.MemberName.Equals("gas"))
239  {
240  node.Expression = nestedMa.Expression;
241  node.MsgValue = nestedMa.MemberName.Equals("value") ? nestedFunctionCall.Arguments[0] : node.MsgValue;
242  node.MsgGas = nestedMa.MemberName.Equals("gas") ? nestedFunctionCall.Arguments[0] : node.MsgGas;
243  }
244  }
245  }
246  }
247  }
248  }
249  return base.Visit(node);
250  }

263  private void InsertStatementBeforeImpl(Block block, Statement beforeThis, Statement stmt)
264  {
265  for (int i = 0; i < block.Statements.Count; ++i)
266  {
267  Statement currentStmt = block.Statements[i];
268  if (currentStmt == beforeThis)
269  {
270  block.Statements.Insert(i, stmt);
271  }
272  else if (currentStmt is Block subBlock)
273  {
274  InsertStatementBeforeImpl(subBlock, beforeThis, stmt);
275  }
276  else if (currentStmt is IfStatement ifStmt)
277  {
278  if (ifStmt.TrueBody is Block)
279  {
280  InsertStatementBeforeImpl(ifStmt.TrueBody as Block, beforeThis, stmt);
281  }
282  else if (ifStmt.TrueBody == beforeThis)
283  {
284  ifStmt.TrueBody = CreateBlockFromTwoStatements(stmt, beforeThis);
285  }
286 
287  if (ifStmt.FalseBody is Block)
288  {
289  InsertStatementBeforeImpl(ifStmt.FalseBody as Block, beforeThis, stmt);
290  }
291  else if (ifStmt.FalseBody == beforeThis)
292  {
293  ifStmt.FalseBody = CreateBlockFromTwoStatements(stmt, beforeThis);
294  }
295  }
296  else if (currentStmt is WhileStatement whileStmt)
297  {
298  if (whileStmt.Body is Block)
299  {
300  InsertStatementBeforeImpl(whileStmt.Body as Block, beforeThis, stmt);
301  }
302  else if (whileStmt.Body == beforeThis)
303  {
304  whileStmt.Body = CreateBlockFromTwoStatements(stmt, beforeThis);
305  }
306  }
307  else if (currentStmt is ForStatement forStmt)
308  {
309  if (forStmt.Body is Block)
310  {
311  InsertStatementBeforeImpl(forStmt.Body as Block, beforeThis, stmt);
312  }
313  else if (forStmt.Body == beforeThis)
314  {
315  forStmt.Body = CreateBlockFromTwoStatements(stmt, beforeThis);
316  }
317  }
318  else if (currentStmt is DoWhileStatement doStmt)
319  {
320  if (doStmt.Body is Block)
321  {
322  InsertStatementBeforeImpl(doStmt.Body as Block, beforeThis, stmt);
323  }
324  else if (doStmt.Body == beforeThis)
325  {
326  doStmt.Body = CreateBlockFromTwoStatements(stmt, beforeThis);
327  }
328  }
329  }
330  }
