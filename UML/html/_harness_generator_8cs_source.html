27  {
28  foreach (ContractDefinition contract in context.ContractDefinitions)
29  {
30  if (contract.ContractKind == EnumContractKind.LIBRARY &&
31  contract.Name.Equals("VeriSol"))
32  {
33  continue;
34  }
35 
36  Dictionary<int, BoogieExpr> houdiniVarMap =
37  HoudiniHelper.GenerateHoudiniVarMapping(contract, context);
38  GenerateHoudiniVarsForContract(contract, houdiniVarMap);
39  GenerateBoogieHarnessForContract(contract, houdiniVarMap);
40  }
41  }
42 
43  GenerateModifiers();
44 
45  foreach (ContractDefinition contract in context.ContractDefinitions)
46  {
47  if (contract.ContractKind == EnumContractKind.LIBRARY &&
48  contract.Name.Equals("VeriSol"))
49  {
50  continue;
51  }
52 
53  GenerateCorralChoiceProcForContract(contract);
54  GenerateCorralHarnessForContract(contract);
55  }

93  private void GenerateBoogieHarnessForContract(ContractDefinition contract, Dictionary<int, BoogieExpr> houdiniVarMap)
94  {
95  string harnessName = "BoogieEntry_" + contract.Name;
96  List<BoogieVariable> inParams = new List<BoogieVariable>();
97  List<BoogieVariable> outParams = new List<BoogieVariable>();
98  BoogieProcedure harness = new BoogieProcedure(harnessName, inParams, outParams);
99  context.Program.AddDeclaration(harness);
100 
101  List<BoogieVariable> localVars = TransUtils.CollectLocalVars(new List<ContractDefinition>() { contract }, context);
102  localVars.Add(new BoogieLocalVariable(new BoogieTypedIdent("tmpNow", BoogieType.Int)));
103  BoogieStmtList harnessBody = new BoogieStmtList();
104  harnessBody.AddStatement(new BoogieAssumeCmd(new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.GE, new BoogieIdentifierExpr("now"), new BoogieLiteralExpr(0))));
105  harnessBody.AddStatement(GenerateDynamicTypeAssumes(contract));
106  GenerateConstructorCall(contract).ForEach(x => harnessBody.AddStatement(x));
107  if (context.TranslateFlags.ModelReverts)
108  {
109  BoogieExpr assumePred = new BoogieUnaryOperation(BoogieUnaryOperation.Opcode.NOT, new BoogieIdentifierExpr("revert"));
110  if (context.TranslateFlags.InstrumentGas)
111  {
112  assumePred = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.AND, assumePred, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.GE, new BoogieIdentifierExpr("gas"), new BoogieLiteralExpr(0)));
113  }
114 
115  harnessBody.AddStatement(new BoogieAssumeCmd(assumePred));
116  }
117  harnessBody.AddStatement(GenerateWhileLoop(contract, houdiniVarMap, localVars));
118  BoogieImplementation harnessImpl = new BoogieImplementation(harnessName, inParams, outParams, localVars, harnessBody);
119  context.Program.AddDeclaration(harnessImpl);
120  }

122  private BoogieAssumeCmd GenerateDynamicTypeAssumes(ContractDefinition contract)
123  {
124  BoogieExpr assumeLhs = new BoogieMapSelect(new BoogieIdentifierExpr("DType"), new BoogieIdentifierExpr("this"));
125 
126  List<ContractDefinition> subtypes = new List<ContractDefinition>(context.GetSubTypesOfContract(contract));
127  Debug.Assert(subtypes.Count > 0);
128 
129  BoogieExpr assumeExpr0 = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, assumeLhs,
130  new BoogieIdentifierExpr(subtypes[0].Name));
131  var assumeExpr = assumeExpr0;
132 
133  for (int i = 1; i < subtypes.Count; ++i)
134  {
135  BoogieExpr rhs = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, assumeLhs,
136  new BoogieIdentifierExpr(subtypes[i].Name));
137  assumeExpr = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.OR, assumeExpr, rhs);
138  }
139 
140  if (context.TranslateFlags.OmitAssumeFalseForDynDispatch)
141  {
142  return new BoogieAssumeCmd(new BoogieLiteralExpr(true)); //assumeExpr0);
143  }
144  return new BoogieAssumeCmd(assumeExpr);
145  }

148  private List<BoogieCmd> GenerateConstructorCall(ContractDefinition contract)
149  {
150  List<BoogieCmd> localStmtList = new List<BoogieCmd>();
151  string callee = TransUtils.GetCanonicalConstructorName(contract);
152  List<BoogieExpr> inputs = new List<BoogieExpr>()
153  {
154  new BoogieIdentifierExpr("this"),
155  new BoogieIdentifierExpr("msgsender_MSG"),
156  new BoogieIdentifierExpr("msgvalue_MSG"),
157  };
158  if (context.IsConstructorDefined(contract))
159  {
160  FunctionDefinition ctor = context.GetConstructorByContract(contract);
161  foreach (VariableDeclaration param in ctor.Parameters.Parameters)
162  {
163  string name = TransUtils.GetCanonicalLocalVariableName(param, context);
164  inputs.Add(new BoogieIdentifierExpr(name));
165 
166  if (param.TypeName is ArrayTypeName array)
167  {
168  localStmtList.Add(new BoogieCallCmd(
169  "FreshRefGenerator",
170  new List<BoogieExpr>(), new List<BoogieIdentifierExpr>() {new BoogieIdentifierExpr(name)}));
171  }
172  }
173  }
174 
175  if (context.TranslateFlags.InstrumentGas)
176  {
177  var gasVar = new BoogieIdentifierExpr("gas");
178  localStmtList.Add(new BoogieAssignCmd(gasVar, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.SUB, gasVar, new BoogieLiteralExpr(TranslatorContext.CREATE_GAS_COST))));
179  }
180 
181  localStmtList.Add(new BoogieCallCmd(callee, inputs, null));
182  return localStmtList;
183  }

185  private BoogieWhileCmd GenerateWhileLoop(ContractDefinition contract, Dictionary<int, BoogieExpr> houdiniVarMap, List<BoogieVariable> localVars)
186  {
187  // havoc all local variables except `this'
188  BoogieStmtList body = GenerateHavocBlock(contract, localVars);
189 
190  // generate the choice block
191  body.AddStatement(TransUtils.GenerateChoiceBlock(new List<ContractDefinition>() { contract }, context));
192 
193  // generate candidate invariants for Houdini
194  List<BoogiePredicateCmd> candidateInvs = new List<BoogiePredicateCmd>();
195  foreach (int id in houdiniVarMap.Keys)
196  {
197  BoogieIdentifierExpr houdiniVar = new BoogieIdentifierExpr(GetHoudiniVarName(id, contract));
198  BoogieExpr candidateInv = houdiniVarMap[id];
199  BoogieExpr invExpr = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.IMP, houdiniVar, candidateInv);
200  BoogieLoopInvCmd invCmd = new BoogieLoopInvCmd(invExpr);
201  candidateInvs.Add(invCmd);
202  }
203 
204  // add the contract invariant if present
205  if (contractInvariants.ContainsKey(contract.Name))
206  {
207  contractInvariants[contract.Name].ForEach(x => candidateInvs.Add(new BoogieLoopInvCmd(x)));
208  }
209 
210  return new BoogieWhileCmd(new BoogieLiteralExpr(true), body, candidateInvs);
211  }

213  private BoogieStmtList GenerateHavocBlock(ContractDefinition contract, List<BoogieVariable> localVars)
214  {
215  BoogieStmtList stmtList = new BoogieStmtList();
216  foreach (BoogieVariable localVar in localVars)
217  {
218  string varName = localVar.TypedIdent.Name;
219  if (!varName.Equals("this"))
220  {
221  stmtList.AddStatement(new BoogieHavocCmd(new BoogieIdentifierExpr(varName)));
222  }
223  }
224 
225  if (context.TranslateFlags.InstrumentGas)
226  {
227  TransUtils.havocGas(stmtList);
228  }
229 
230  var nowVar = new BoogieIdentifierExpr("now");
231  var tmpNowVar = new BoogieIdentifierExpr("tmpNow");
232  stmtList.AddStatement(new BoogieAssignCmd(tmpNowVar, nowVar));
233  stmtList.AddStatement(new BoogieHavocCmd(nowVar));
234  stmtList.AddStatement(new BoogieAssumeCmd(new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.GT, nowVar, tmpNowVar)));
235  stmtList.AddStatement(new BoogieAssumeCmd(new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ, new BoogieIdentifierExpr("msgsender_MSG"), new BoogieIdentifierExpr("null"))));
236  foreach (var contractDef in context.ContractDefinitions)
237  {
238  BoogieIdentifierExpr contractIdent = new BoogieIdentifierExpr(contractDef.Name);
239  stmtList.AddStatement(new BoogieAssumeCmd(new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ,
240  new BoogieMapSelect(new BoogieIdentifierExpr("DType"), new BoogieIdentifierExpr("msgsender_MSG")),
241  contractIdent)));
242  }
243 
244  stmtList.AddStatement((new BoogieAssignCmd(new BoogieMapSelect(new BoogieIdentifierExpr("Alloc"), new BoogieIdentifierExpr("msgsender_MSG")), new BoogieLiteralExpr(true))));
245 
246  return stmtList;
247  }

255  private void GenerateCorralChoiceProcForContract(ContractDefinition contract)
256  {
257  string procName = "CorralChoice_" + contract.Name;
258  List<BoogieVariable> inParams = new List<BoogieVariable>()
259  {

261  };

356  private List<BoogieVariable> GetParamsOfFunction(FunctionDefinition funcDef)
357  {
358  List<BoogieVariable> parameters = new List<BoogieVariable>();
359 
360  var inpParamCount = 0;
361  foreach (VariableDeclaration param in funcDef.Parameters.Parameters)
362  {
363  string name = $"__arg1_{inpParamCount++}_" + funcDef.Name;
364  if (!string.IsNullOrEmpty(param.Name))
365  {
366  name = TransUtils.GetCanonicalLocalVariableName(param, context);
367  }
368  BoogieType type = TransUtils.GetBoogieTypeFromSolidityTypeName(param.TypeName);
369  BoogieVariable localVar = new BoogieLocalVariable(new BoogieTypedIdent(name, type));
370  parameters.Add(localVar);
371  }
372 
373  var retParamCount = 0;
374  foreach (VariableDeclaration param in funcDef.ReturnParameters.Parameters)
375  {
376  string name = $"__ret1_{retParamCount++}_" + funcDef.Name;
377 
378  if (!string.IsNullOrEmpty(param.Name))
379  {
380  name = TransUtils.GetCanonicalLocalVariableName(param, context);
381  }
382  BoogieType type = TransUtils.GetBoogieTypeFromSolidityTypeName(param.TypeName);
383  BoogieVariable localVar = new BoogieLocalVariable(new BoogieTypedIdent(name, type));
384  parameters.Add(localVar);
385  }
386 
387  return parameters;
