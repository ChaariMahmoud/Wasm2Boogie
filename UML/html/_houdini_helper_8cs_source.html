12  // returns a map from integer id to an atomic Boogie predicate
13  public static Dictionary<int, BoogieExpr> GenerateHoudiniVarMapping(ContractDefinition contract, TranslatorContext context)
14  {
15  Dictionary<int, BoogieExpr> ret = new Dictionary<int, BoogieExpr>();
16  HashSet<VariableDeclaration> stateVars = context.GetVisibleStateVarsByContract(contract);
17 
18  // collect all state variables of type address
19  List<VariableDeclaration> addressVariables = new List<VariableDeclaration>();
20  foreach (VariableDeclaration stateVar in stateVars)
21  {
22  if (stateVar.TypeName is ElementaryTypeName elementaryType)
23  {
24  if (elementaryType.TypeDescriptions.TypeString.Equals("address") ||
25  elementaryType.TypeDescriptions.TypeString.Equals("address payable"))
26  {
27  addressVariables.Add(stateVar);
28  }
29  }
30  }
31 
32  int id = 0;
33 
34  // equaility and disequality to null
35  foreach (VariableDeclaration addressVar in addressVariables)
36  {
37  BoogieExpr lhs = GetBoogieExprOfStateVar(addressVar, context);
38  BoogieExpr rhs = new BoogieIdentifierExpr("null");
39  BoogieExpr equality = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, lhs, rhs);
40  ret[++id] = equality;
41  BoogieExpr disequality = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ, lhs, rhs);
42  ret[++id] = disequality;
43  }
44 
45  // pair-wise equality and disequality
46  for (int i = 0; i < addressVariables.Count; ++i)
47  {
48  BoogieExpr lhs = GetBoogieExprOfStateVar(addressVariables[i], context);
49  for (int j = i + 1; j < addressVariables.Count; ++j)
50  {
51  BoogieExpr rhs = GetBoogieExprOfStateVar(addressVariables[j], context);
52  BoogieExpr equality = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, lhs, rhs);
53  ret[++id] = equality;
54  BoogieExpr disequality = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ, lhs, rhs);
55  ret[++id] = disequality;
56  }
57  }
58 
59  // PrintHoudiniCandidateMap(ret);
60  return ret;
61  }
