59  private List<CFGNode> SplitControlFlow(int num)

100  public override bool Visit(BinaryOperation astNode)
101  {
102  switch(astNode.Operator)
103  {
104  case "==":
105  case "!=":
106  case ">=":
107  case "<=":
108  case ">":
109  case "<":
110  currentNode.Block.SetPredicate(astNode);
111  break;
112  case "||":
113  {
114  AppendControlFlow(astNode.LeftExpression);
115  List<CFGNode> nodes = SplitControlFlow(2);
116  nodes[1] = CreateControlFlow(nodes[1], astNode.RightExpression);
117  MergeControlFlow(nodes, nodes[0]);
118  }
119  break;
120  case "&&":
121  {
122  AppendControlFlow(astNode.LeftExpression);
123  List<CFGNode> nodes = SplitControlFlow(2);
124  nodes[0] = CreateControlFlow(nodes[0], astNode.RightExpression);
125  MergeControlFlow(nodes, nodes[1]);
126  }
127  break;
128  default:
129  throw new SystemException($"Unknown operator: {astNode.Operator}");
130  }
131  return false;
132  }

148  public override bool Visit(IfStatement astNode)
149  {
150  if (astNode.Condition is Identifier)
151  {
152  Identifier identifier = (Identifier)astNode.Condition;
153  currentNode.Block.SetPredicate(identifier);
154  }
155  else if (astNode.Condition is BinaryOperation)
156  {
157  astNode.Condition.Accept(this);
158  }
159  else

162  }
163 
164  List<CFGNode> nodes = SplitControlFlow(2);
165  nodes[0] = CreateControlFlow(nodes[0], astNode.TrueBody);
166  if (astNode.FalseBody != null)
167  {
168  nodes[1] = CreateControlFlow(nodes[1], astNode.FalseBody);
169  MergeControlFlow(nodes);
170  }
171  else

173  MergeControlFlow(nodes, nodes[1]);
174  }
