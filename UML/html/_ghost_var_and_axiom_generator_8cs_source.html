347  private void GenerateStructConstructors(ContractDefinition contract, StructDefinition structDefn)
348  {
349  // generate the internal one without base constructors
350  string procName = structDefn.CanonicalName + "_ctor";
351  List<BoogieVariable> inParams = new List<BoogieVariable>();
352  inParams.AddRange(TransUtils.GetDefaultInParams());
353  foreach(var member in structDefn.Members)
354  {
355  Debug.Assert(!member.TypeDescriptions.IsStruct(), "Do no handle nested structs yet!");
356  var formalType = TransUtils.GetBoogieTypeFromSolidityTypeName(member.TypeName);
357  var formalName = member.Name;
358  inParams.Add(new BoogieFormalParam(new BoogieTypedIdent(formalName, formalType)));
359  }
360 
361  List<BoogieVariable> outParams = new List<BoogieVariable>();
362  List<BoogieAttribute> attributes = new List<BoogieAttribute>();
363  if (context.TranslateFlags.GenerateInlineAttributes)
364  {
365  attributes.Add(new BoogieAttribute("inline", 1));
366  };
367  BoogieProcedure procedure = new BoogieProcedure(procName, inParams, outParams, attributes);
368  context.Program.AddDeclaration(procedure);
369 
370  List<BoogieVariable> localVars = new List<BoogieVariable>();
371  BoogieStmtList procBody = new BoogieStmtList();
372 
373  foreach (var member in structDefn.Members)
374  {
375  //f[this] = f_arg
376  Debug.Assert(!member.TypeDescriptions.IsStruct(), "Do no handle nested structs yet!");
377  var mapName = member.Name + "_" + structDefn.CanonicalName;
378  var formalName = member.Name;
379  var mapSelectExpr = new BoogieMapSelect(new BoogieIdentifierExpr(mapName), new BoogieIdentifierExpr("this"));
380  procBody.AddStatement(new BoogieAssignCmd(mapSelectExpr, new BoogieIdentifierExpr(member.Name)));
381  }
382 
383  BoogieImplementation implementation = new BoogieImplementation(procName, inParams, outParams, localVars, procBody);
384  context.Program.AddDeclaration(implementation);
385  }

402  private void GenerateGlobalProcedureFresh()
403  {
404  // generate the internal one without base constructors
405  string procName = "FreshRefGenerator";
406  List<BoogieVariable> inParams = new List<BoogieVariable>();
407 
408  var outVar = new BoogieFormalParam(new BoogieTypedIdent("newRef", BoogieType.Ref));
409  List<BoogieVariable> outParams = new List<BoogieVariable>()
410  {
411  outVar
412  };
413  List<BoogieAttribute> attributes = new List<BoogieAttribute>();
414  if (context.TranslateFlags.GenerateInlineAttributes)
415  {
416  attributes.Add(new BoogieAttribute("inline", 1));
417  };
418  BoogieProcedure procedure = new BoogieProcedure(procName, inParams, outParams, attributes);
419  context.Program.AddDeclaration(procedure);
420 
421  List<BoogieVariable> localVars = new List<BoogieVariable>();
422  BoogieStmtList procBody = new BoogieStmtList();
423 
424  var outVarIdentifier = new BoogieIdentifierExpr("newRef");
425  BoogieIdentifierExpr allocIdentExpr = new BoogieIdentifierExpr("Alloc");
426  // havoc tmp;
427  procBody.AddStatement(new BoogieHavocCmd(outVarIdentifier));
428  // assume Alloc[tmp] == false;
429  BoogieMapSelect allocMapSelect = new BoogieMapSelect(allocIdentExpr, outVarIdentifier);
430  BoogieExpr allocAssumeExpr = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, allocMapSelect, new BoogieLiteralExpr(false));
431  procBody.AddStatement(new BoogieAssumeCmd(allocAssumeExpr));
432  // Alloc[tmp] := true;
433  procBody.AddStatement(new BoogieAssignCmd(allocMapSelect, new BoogieLiteralExpr(true)));
434  // assume tmp != null
435  procBody.AddStatement(new BoogieAssumeCmd(
436  new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ, outVarIdentifier, new BoogieIdentifierExpr("null"))));
437 
438  BoogieImplementation implementation = new BoogieImplementation(procName, inParams, outParams, localVars, procBody);
439  context.Program.AddDeclaration(implementation);
440  }

442  private void GenerateGlobalProcedureAllocMany()
443  {
444  if (context.TranslateFlags.LazyNestedAlloc)
445  return;
446 
447  // generate the internal one without base constructors
448  string procName = "HavocAllocMany";
449  List<BoogieVariable> inParams = new List<BoogieVariable>();
450  List<BoogieVariable> outParams = new List<BoogieVariable>();
451  List<BoogieAttribute> attributes = new List<BoogieAttribute>();
452  if (context.TranslateFlags.GenerateInlineAttributes)
453  {
454  attributes.Add(new BoogieAttribute("inline", 1));
455  };
456  BoogieProcedure procedure = new BoogieProcedure(procName, inParams, outParams, attributes);
457  context.Program.AddDeclaration(procedure);
458 
459  var oldAlloc = new BoogieLocalVariable(new BoogieTypedIdent("oldAlloc", new BoogieMapType(BoogieType.Ref, BoogieType.Bool)));
460  List<BoogieVariable> localVars = new List<BoogieVariable>() {oldAlloc};
461  BoogieStmtList procBody = new BoogieStmtList();
462  BoogieIdentifierExpr oldAllocIdentExpr = new BoogieIdentifierExpr("oldAlloc");
463  BoogieIdentifierExpr allocIdentExpr = new BoogieIdentifierExpr("Alloc");
464  // oldAlloc = Alloc
465  procBody.AddStatement(new BoogieAssignCmd(oldAllocIdentExpr, allocIdentExpr));
466  // havoc Alloc
467  procBody.AddStatement(new BoogieHavocCmd(allocIdentExpr));
468  // assume forall i:ref oldAlloc[i] ==> alloc[i]
469  var qVar = QVarGenerator.NewQVar(0, 0);
470  BoogieMapSelect allocMapSelect = new BoogieMapSelect(allocIdentExpr, qVar);
471  BoogieMapSelect oldAllocMapSelect = new BoogieMapSelect(oldAllocIdentExpr, qVar);
472  BoogieExpr allocAssumeExpr = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.IMP, oldAllocMapSelect, allocMapSelect);
473  procBody.AddStatement(new BoogieAssumeCmd(new BoogieQuantifiedExpr(true, new List<BoogieIdentifierExpr>() {qVar}, new List<BoogieType>() { BoogieType.Ref }, allocAssumeExpr)));
474 
475  BoogieImplementation implementation = new BoogieImplementation(procName, inParams, outParams, localVars, procBody);
476  context.Program.AddDeclaration(implementation);
477  }

497  private BoogieAxiom GenerateConstToRefAxiom()
498  {
499 
500  var qVar1 = QVarGenerator.NewQVar(0, 0);
501  var qVar2 = QVarGenerator.NewQVar(0, 1);
502  var eqVar12 = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, qVar1, qVar2);
503  var qVar1Func = new BoogieFuncCallExpr("ConstantToRef", new List<BoogieExpr>() { qVar1 });
504  var qVar2Func = new BoogieFuncCallExpr("ConstantToRef", new List<BoogieExpr>() { qVar2 });
505  var eqFunc12 = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ, qVar1Func, qVar2Func);
506  var bodyExpr = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.OR, eqVar12, eqFunc12);
507  var triggers = new List<BoogieExpr>() { qVar1Func, qVar2Func };
508 
509  // forall q1:int, q2:int :: q1 == q2 || ConstantToRef(q1) != ConstantToRef(q2)
510  var qExpr = new BoogieQuantifiedExpr(true, new List<BoogieIdentifierExpr>() { qVar1, qVar2 }, new List<BoogieType>() { BoogieType.Int, BoogieType.Int }, bodyExpr, triggers);
511 
512  return new BoogieAxiom(qExpr);
513  }
514  private BoogieAxiom GenerateKeccakAxiom()
515  {
516 
517  var qVar1 = QVarGenerator.NewQVar(0, 0);
518  var qVar2 = QVarGenerator.NewQVar(0, 1);
519  var eqVar12 = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, qVar1, qVar2);
520  var qVar1Func = new BoogieFuncCallExpr("keccak256", new List<BoogieExpr>() { qVar1 });
521  var qVar2Func = new BoogieFuncCallExpr("keccak256", new List<BoogieExpr>() { qVar2 });
522  var eqFunc12 = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.NEQ, qVar1Func, qVar2Func);
523  var bodyExpr = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.OR, eqVar12, eqFunc12);
524  var triggers = new List<BoogieExpr>() { qVar1Func, qVar2Func };
525 
526  // forall q1:int, q2:int :: q1 == q2 || keccak256(q1) != keccak256(q2)
527  var qExpr = new BoogieQuantifiedExpr(true, new List<BoogieIdentifierExpr>() { qVar1, qVar2 }, new List<BoogieType>() { BoogieType.Int, BoogieType.Int }, bodyExpr, triggers);
528 
529  return new BoogieAxiom(qExpr);
530  }
531  private BoogieAxiom GenerateAbiEncodePackedAxiomOneArg()
532  {
533

678  private void GenerateMemoryVariablesForMapping(Mapping mapping, HashSet<KeyValuePair<BoogieType, BoogieType>> generatedTypes)
679  {
680  BoogieType boogieKeyType = TransUtils.GetBoogieTypeFromSolidityTypeName(mapping.KeyType);
681  BoogieType boogieValType = null;
682  if (mapping.ValueType is Mapping submapping)
683  {

685  GenerateMemoryVariablesForMapping(submapping, generatedTypes);
686  }
687  else if (mapping.ValueType is ArrayTypeName array)

702  }
