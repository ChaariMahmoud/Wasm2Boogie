31  public static BoogieType InferExprTypeFromTypeString(string typeString)
32  {
33  //TODO: use TypeDescriptions::IsInt/IsStruct etc., but need typeDescription instead of typeString
34  if (IsArrayTypeString(typeString))
35  {
36  return BoogieType.Ref;
37  }
38  else if (IsMappingTypeString(typeString))
39  {
40  return BoogieType.Ref;
41  }
42  else if (typeString.Equals("address") || typeString.Equals("address payable"))
43  {
44  return BoogieType.Ref;
45  }
46  else if (typeString.Equals("bool"))
47  {
48  return BoogieType.Bool;
49  }
50  else if (typeString.StartsWith("uint") && !typeString.Contains("["))
51  {
52  return BoogieType.Int;
53  }
54  else if (typeString.StartsWith("int") && !typeString.Contains("["))
55  {
56  return BoogieType.Int;
57  }
58  else if (typeString.StartsWith("byte") && !typeString.Contains("["))
59  {
60  return BoogieType.Int;
61  }
62  else if (typeString.StartsWith("contract "))
63  {
64  return BoogieType.Ref;
65  }
66  else if (typeString.StartsWith("struct "))
67  {
68  return BoogieType.Ref;
69  }
70  else if (typeString.Equals("string") || typeString.StartsWith("string "))
71  {
72  return BoogieType.Int; //we think of string as its hash
73  }
74  else if (typeString.StartsWith("literal_string "))
75  {
76  return BoogieType.Int; //we think of string as its hash
77  }
78  else
79  {
80  throw new SystemException($"Cannot infer from type string: {typeString}");
81  }
82  }
