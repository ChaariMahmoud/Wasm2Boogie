144  BoogieStmtList dupAndReplaceStmList(BoogieStmtList stmtList)
145  {
146  if (stmtList == null)
147  return null;
148 
149  BoogieStmtList newList = new BoogieStmtList();
150  stmtList.BigBlocks[0].SimpleCmds.ForEach(c => generateRevertLogicForCmd(c, newList, isFail, inHarness));
151  return newList;
152  }
153 
154  if (cmd is BoogieAssignCmd assignCmd)
155  {
156  parent.AddStatement(new BoogieAssignCmd(dupAndReplaceExpr(assignCmd.Lhs, isFail, inHarness),
157  dupAndReplaceExpr(assignCmd.Rhs, isFail, inHarness)));
158  }
159  else if (cmd is BoogieCallCmd callCmd)
160  {
161  string calleeName = callCmd.Callee;
162  bool emitCheckRevertLogic = false;
163  if (!isHarnessProcudure(calleeName) && procsWiltImplNames.Contains(calleeName))
164  {
165  if (!inHarness || !isPublic(proceduresInProgram[calleeName]))
166  {
167  emitCheckRevertLogic = !inHarness && !catchesExceptions(calleeName);
168  calleeName = calleeName + (isFail ? "__fail" : "__success");
169  }
170  }
171 
172  var newIns = callCmd.Ins != null ? dupAndReplaceExprList(callCmd.Ins) : null;
173  var newOuts = callCmd.Outs?.Select(e => (BoogieIdentifierExpr) dupAndReplaceExpr(e, isFail, inHarness)).ToList();
174  parent.AddStatement(new BoogieCallCmd(calleeName, newIns, newOuts));
175 
176  if (emitCheckRevertLogic)
177  {
178  BoogieStmtList thenBody = new BoogieStmtList();
179  thenBody.AddStatement(new BoogieReturnCmd());
180 
181  parent.AddStatement(new BoogieIfCmd(new BoogieIdentifierExpr("revert"), thenBody, null));
182  }
183  }
184  else if (cmd is BoogieAssertCmd assertCmd)
185  {
186  if (!isFail)
187  {
188  parent.AddStatement(new BoogieAssertCmd(dupAndReplaceExpr(assertCmd.Expr, false, inHarness), assertCmd.Attributes));
189  }
190  }
191  else if (cmd is BoogieAssumeCmd assumeCmd)
192  {
193  parent.AddStatement(new BoogieAssumeCmd(dupAndReplaceExpr(assumeCmd.Expr, isFail, inHarness)));
194  }
195  else if (cmd is BoogieLoopInvCmd loopInvCmd)
196  {
197  parent.AddStatement(new BoogieLoopInvCmd(dupAndReplaceExpr(loopInvCmd.Expr, isFail, inHarness)));
198  }
199  else if (cmd is BoogieReturnExprCmd returnExprCmd)
200  {
201  // This one does not seem to be used.
202  parent.AddStatement(new BoogieReturnExprCmd(dupAndReplaceExpr(returnExprCmd.Expr, isFail, inHarness)));
203  }
204  else if (cmd is BoogieHavocCmd havocCmd)
205  {
206  parent.AddStatement(new BoogieHavocCmd(havocCmd.Vars.Select(id => (BoogieIdentifierExpr) dupAndReplaceExpr(id, isFail, inHarness)).ToList()));
207  }
208  else if (cmd is BoogieIfCmd ifCmd)
209  {
210  parent.AddStatement(new BoogieIfCmd(dupAndReplaceExpr(ifCmd.Guard, isFail, inHarness),
211  dupAndReplaceStmList(ifCmd.ThenBody),
212  dupAndReplaceStmList(ifCmd.ElseBody)));
213  }
214  else if (cmd is BoogieWhileCmd whileCmd)
215  {
216  var body = dupAndReplaceStmList(whileCmd.Body);
217  BoogieStmtList invsAsStmtList = new BoogieStmtList();
218  whileCmd.Invariants.ForEach(i => generateRevertLogicForCmd(i, invsAsStmtList, isFail, inHarness));
219  List<BoogiePredicateCmd> invs = invsAsStmtList.BigBlocks[0].SimpleCmds.Select(c => (BoogiePredicateCmd) c).ToList();
220  parent.AddStatement( new BoogieWhileCmd(dupAndReplaceExpr(whileCmd.Guard, isFail, inHarness), body, invs));
221  }
222  else
223  {
224  parent.AddStatement(cmd);
225  }

271  public void Generate()
272  {
273  // Collect Global Variables.
274  HashSet<BoogieGlobalVariable> globals = new HashSet<BoogieGlobalVariable>();
275  foreach (var decl in context.Program.Declarations)
276  {
277  if (decl is BoogieGlobalVariable)
278  {
279  var g = (BoogieGlobalVariable)decl;
280 
281  if (mustHaveShadow(g.Name))
282  globals.Add(g);
283  }
284  }
285 
286  // Generate shadow state.
287  foreach (var g in globals)
288  {
289  var varName = g.TypedIdent.Name;
290  BoogieTypedIdent shadowGlobal = new BoogieTypedIdent("__tmp__" + varName, g.TypedIdent.Type);
291  BoogieGlobalVariable shadowGlobalDecl = new BoogieGlobalVariable(shadowGlobal);
292 
293  context.Program.AddDeclaration(shadowGlobalDecl);
294  shadowGlobals.Add(varName, shadowGlobalDecl);
295  }
296 
297  // Duplicate and rename methods.
298  Dictionary<string, BoogieImplementation> proceduresWithImpl = new Dictionary<string, BoogieImplementation>();
299  procsWiltImplNames = proceduresWithImpl.Keys;
300  foreach (var decl in context.Program.Declarations)
301  {
302  if (decl is BoogieImplementation)
303  {
304  var boogieImpl = ((BoogieImplementation) decl);
305  if (proceduresInProgram.ContainsKey(boogieImpl.Name))
306  {
307  proceduresWithImpl.Add(boogieImpl.Name, boogieImpl);
308  }
309  }
310  }
311 
312  Dictionary<string, BoogieProcedure> failProcedures = new Dictionary<string, BoogieProcedure>();
313  foreach (var implPair in proceduresWithImpl)
314  {
315  BoogieProcedure proc = proceduresInProgram[implPair.Key];
316 
317  if (isPublic(proc) || isConstructor(proc.Name))
318  {
319  // If public maintain original definition as the wrapper.
320  BoogieProcedure successVersion = duplicateProcedure(proc, "success", true);
321  //BoogieImplementation successImpl = duplicateImplementation(implPair.Value, "success");
322 
323  context.Program.AddDeclaration(successVersion);
324  //context.Program.AddDeclaration(successImpl);
325 
326  BoogieProcedure failVersion = duplicateProcedure(proc, "fail", true);
327 
328  context.Program.AddDeclaration(failVersion);
329 
330  // Use original name of the procedure.
331  failProcedures.Add(implPair.Key, failVersion);
332  }
333  else if (!isHarnessProcudure(proc.Name))
334  {
335  // Otherwise reuse original definition/impl as the "successful" method.
336  BoogieProcedure failVersion = duplicateProcedure(proc, "fail");
337 
338  context.Program.AddDeclaration(failVersion);
339 
340  // Use original name of the procedure.
341  failProcedures.Add(implPair.Key, failVersion);
342 
343  // Reuse original node
344  proc.Name = proc.Name + "__success";
345  implPair.Value.Name = implPair.Value.Name + "__success";
346  }
347  }
348 
349  // Create implementations for failing methods.
350  foreach (var failProcedurePair in failProcedures)
351  {
352  string originalProcName = failProcedurePair.Key;
353  BoogieProcedure proc = failProcedurePair.Value;
354 
355  var originalImpl = proceduresWithImpl[originalProcName];
356  if (!originalProcName.Equals("send"))
357  {
358  context.Program.AddDeclaration(createFailImplementation(proc.Name, originalImpl));
359  context.Program.AddDeclaration(createSuccessImplementation(originalProcName + "__success", originalImpl));
360  }
361  else
362  {
363  context.Program.AddDeclaration(CreateSendFail());
364  context.Program.AddDeclaration(CreateSendSucess());
365  }
366 
367  // Remove original implementation for non-public methods
368  if (!isPublic(proceduresInProgram[originalProcName]) && !isConstructor(originalProcName))
369  {
370  context.Program.Declarations.Remove(originalImpl);
371  }
372  }
373 
374  foreach (var implPair in proceduresWithImpl)
375  {
376  // Update non-public methods in harness methods
377  if (isHarnessProcudure(implPair.Key))
378  {
379  context.Program.AddDeclaration(createSuccessImplementation(implPair.Key, implPair.Value));
380 
381  context.Program.Declarations.Remove(implPair.Value);
382  }
383  }
384 
385  // Create wrappers for public methods.
386 
387  foreach (var proc in proceduresInProgram.Values)
388  {
389  if (proceduresWithImpl.ContainsKey(proc.Name) && (isPublic(proc) || isConstructor(proc.Name)))
390  {
391  BoogieImplementation impl = proceduresWithImpl[proc.Name];
392  impl.StructuredStmts = new BoogieStmtList();
393  impl.LocalVars = new List<BoogieVariable>();
394 
395  var exceptionVarName = "__exception";
396  var revertGlobalName = "revert";
397  impl.LocalVars.Add(new BoogieLocalVariable(new BoogieTypedIdent(exceptionVarName, BoogieType.Bool)));
398 
399  var stmtList = impl.StructuredStmts;
400  stmtList.AddStatement(new BoogieHavocCmd(new BoogieIdentifierExpr(exceptionVarName)));
401  stmtList.AddStatement(new BoogieAssignCmd(new BoogieIdentifierExpr(revertGlobalName), new BoogieLiteralExpr(false)));
402 
403  // Call Successful version.
404  BoogieStmtList successCallStmtList = new BoogieStmtList();
405  successCallStmtList.AddStatement(new BoogieCallCmd(impl.Name + "__success",
406  impl.InParams.Select(inParam => (BoogieExpr)new BoogieIdentifierExpr(inParam.Name)).ToList(),
407  impl.OutParams.Select(outParam => new BoogieIdentifierExpr(outParam.Name)).ToList()));
408  BoogieExpr successAssumePred = new BoogieUnaryOperation(BoogieUnaryOperation.Opcode.NOT, new BoogieIdentifierExpr(revertGlobalName));
409 
410  if (context.TranslateFlags.InstrumentGas)
411  {
412  successAssumePred = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.AND, successAssumePred, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.GE, new BoogieIdentifierExpr("gas"), new BoogieLiteralExpr(0)));
413  }
414  successCallStmtList.AddStatement(new BoogieAssumeCmd(successAssumePred));
415 
416  // Call fail version.
417  BoogieStmtList failCallStmtList = new BoogieStmtList();
418  // Write global variables to temps
419  foreach (var shadowGlobalPair in shadowGlobals)
420  {
421  string origVarName = shadowGlobalPair.Key;
422  string shadowName = shadowGlobalPair.Value.Name;
423  failCallStmtList.AddStatement(new BoogieAssignCmd(new BoogieIdentifierExpr(shadowName), new BoogieIdentifierExpr(origVarName)));
424  }
425  failCallStmtList.AddStatement(new BoogieCallCmd(impl.Name + "__fail",
426  impl.InParams.Select(inParam => (BoogieExpr)new BoogieIdentifierExpr(inParam.Name)).ToList(),
427  impl.OutParams.Select(outParam => new BoogieIdentifierExpr(outParam.Name)).ToList()));
428  BoogieExpr failAssumePred = new BoogieIdentifierExpr(revertGlobalName);
429 
430  if (context.TranslateFlags.InstrumentGas)
431  {
432  failAssumePred = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.OR, failAssumePred, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.LT, new BoogieIdentifierExpr("gas"), new BoogieLiteralExpr(0)));
433  }
434  failCallStmtList.AddStatement(new BoogieAssumeCmd(failAssumePred));
435 
436  stmtList.AddStatement(new BoogieIfCmd(new BoogieIdentifierExpr(exceptionVarName), failCallStmtList, successCallStmtList));
437  }
438  }
