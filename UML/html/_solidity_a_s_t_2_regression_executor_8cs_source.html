80  private bool TextCompare(string originalText, string astText)
81  {

88  string[] astWords = astText.Split(separator, StringSplitOptions.RemoveEmptyEntries);
89

93  "CONTRACT", "LIBRARY", "INTERFACE",
94  "PUBLIC", "EXTERNAL", "INTERNAL", "PRIVATE", "DEFAULT",
95  "STORAGE", "MEMORY", "CALLDATA",
96  "PURE", "VIEW", "NONPAYABLE", "PAYABLE",
97  };
98 
99  logger.LogDebug($"Normalized program words: {ConcatStringArray(normalizedWords, " ")}");
100  logger.LogDebug($"AST traversal prog words: {ConcatStringArray(astWords, " ")}");
101 
102  if (normalizedWords.Length != astWords.Length)
103  {
104  return false;
105  }
106  for (int i = 0; i < normalizedWords.Length; ++i)
107  {
108  if (enumKeywords.Contains(astWords[i]))
109  {

111  {
112  return false;
113  }
114  }
115  else if (!normalizedWords[i].Equals(astWords[i]))
116  {
117  return false;
118  }
119  }

121  }
