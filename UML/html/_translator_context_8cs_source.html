355  public void AddFunctionToContract(ContractDefinition contract, FunctionDefinition funcDef)
356  {
357  if (!ContractToFunctionsMap.ContainsKey(contract))
358  {
359  Debug.Assert(!ContractToFuncSigsMap.ContainsKey(contract));
360  ContractToFunctionsMap[contract] = new HashSet<FunctionDefinition>();
361  ContractToFuncSigsMap[contract] = new HashSet<string>();
362  }
363 
364  Debug.Assert(!ContractToFunctionsMap[contract].Contains(funcDef), $"Duplicated function definition: {funcDef.Name}");
365  ContractToFunctionsMap[contract].Add(funcDef);
366 
367  string signature = TransUtils.ComputeFunctionSignature(funcDef);
368  Debug.Assert(!ContractToFuncSigsMap[contract].Contains(signature), $"Duplicated function signature: {signature}");
369  ContractToFuncSigsMap[contract].Add(signature);
370 
371  Debug.Assert(!FunctionToContractMap.ContainsKey(funcDef), $"Duplicated function: {funcDef.Name}");
372  FunctionToContractMap[funcDef] = contract;
373  }
