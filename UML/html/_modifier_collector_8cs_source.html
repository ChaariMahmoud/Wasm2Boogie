13  * Collect all modifier definitions and put them in the translator context.

26  public override bool Visit(ModifierDefinition modifier)
27  {
28  //if (modifier.Parameters.Length() > 0)
29  //{
30  // throw new System.Exception("modifiers with parameters not implemented");
31  //}
32  var modifierInParams = TransUtils.GetDefaultInParams();
33 
34  foreach (var parameter in modifier.Parameters.Parameters)
35  {
36  string name = null;
37  name = TransUtils.GetCanonicalLocalVariableName(parameter, context);
38  BoogieType type = TransUtils.GetBoogieTypeFromSolidityTypeName(parameter.TypeName);
39  modifierInParams.Add(new BoogieFormalParam(new BoogieTypedIdent(name, type)));
40  }
41 
42  Block body = modifier.Body;
43  bool hasPre = false;
44  bool hasPost = false;
45  List<Statement> postlude = new List<Statement>();
46 
47  // this list does not include locals introduced during translation
48  List<BoogieLocalVariable> localVarsDeclared = new List<BoogieLocalVariable>();
49 
50  bool translatingPre = true;
51  foreach (Statement statement in body.Statements)
52  {
53  if (statement is VariableDeclarationStatement varDecls)
54  {
55  foreach (VariableDeclaration varDecl in varDecls.Declarations)
56  {
57  string name = TransUtils.GetCanonicalLocalVariableName(varDecl, context);
58  BoogieType type = TransUtils.GetBoogieTypeFromSolidityTypeName(varDecl.TypeName);
59  // Issue a warning for intXX variables in case /useModularArithemtic option is used:
60  if (context.TranslateFlags.UseModularArithmetic && varDecl.TypeDescriptions.IsInt())
61  {
62  Console.WriteLine($"Warning: signed integer arithmetic is not handled with /useModularArithmetic option");
63  }
64  var boogieVariable = new BoogieLocalVariable(new BoogieTypedIdent(name, type));
65  if (translatingPre)
66  localVarsDeclared.Add(boogieVariable); // don't add locals after placeholder
67  }
68 
69  // throw new System.Exception("locals within modifiers not supported");
70  }
71  if (statement is PlaceholderStatement)
72  {
73  translatingPre = false;
74  // only capture those locals declared in the prefix, these are visible to postlude
75  context.AddPreludeLocalsToModifier(modifier.Name, localVarsDeclared);
76 
77  continue;
78  }
79  if (translatingPre)
80  {
81  hasPre = true;
82  }
83  else
84  {
85  hasPost = true;
86  }
87  }
88 
89  var attributes = new List<BoogieAttribute>();
90  attributes.Add(new BoogieAttribute("inline", 1));
91 
92  if (hasPre)
93  {
94  List<BoogieVariable> inParams = new List<BoogieVariable>(modifierInParams);
95  List<BoogieVariable> outParams = new List<BoogieVariable>();
96  outParams.AddRange(localVarsDeclared.Select(x => new BoogieFormalParam(new BoogieTypedIdent("__out_mod_" + x.Name, x.TypedIdent.Type))));
97  BoogieProcedure preludeProc = new BoogieProcedure(modifier.Name + "_pre", inParams, outParams, attributes);
98  context.AddModiferToPreProc(modifier.Name, preludeProc);
99 
100  BoogieImplementation preludeImpl = new BoogieImplementation(modifier.Name + "_pre",
101  inParams, outParams, new List<BoogieVariable>(), new BoogieStmtList());
102  context.AddModiferToPreImpl(modifier.Name, preludeImpl);
103  }
104 
105  if (hasPost)
106  {
107  List<BoogieVariable> inParams = new List<BoogieVariable>(modifierInParams);
108  inParams.AddRange(localVarsDeclared);
109  List<BoogieVariable> outParams = new List<BoogieVariable>();
110  BoogieProcedure postludeProc = new BoogieProcedure(modifier.Name + "_post", inParams, outParams, attributes);
111  context.AddModiferToPostProc(modifier.Name, postludeProc);
112 
113  BoogieImplementation postludeImpl = new BoogieImplementation(modifier.Name + "_post",
114  inParams, outParams, new List<BoogieVariable>(), new BoogieStmtList());
115  context.AddModiferToPostImpl(modifier.Name, postludeImpl);
116  }
117 
118  return false;
119  }
120  }
