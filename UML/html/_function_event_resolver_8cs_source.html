32  private void ResolveFunctions()
33  {
34  ResolutionHelper resolutionHelper = new ResolutionHelper(context);
35  List<ContractDefinition> sortedContracts = resolutionHelper.TopologicalSortByDependency(context.ContractDefinitions);
36 
37  foreach (ContractDefinition contract in sortedContracts)
38  {
39  // create a deep copy
40  List<int> linearizedBaseContractIds = new List<int>(contract.LinearizedBaseContracts);
41  linearizedBaseContractIds.Reverse();
42 
43  foreach (int id in linearizedBaseContractIds)
44  {
45  ContractDefinition baseContract = context.GetASTNodeById(id) as ContractDefinition;
46  Debug.Assert(baseContract != null);
47 
48  if (baseContract == contract)
49  {
50  HashSet<FunctionDefinition> functions = context.GetFuncDefintionsInContract(contract);
51  foreach (FunctionDefinition function in functions)
52  {
53  string signature = TransUtils.ComputeFunctionSignature(function);
54  context.AddFunctionToDynamicType(signature, contract, function);
55  }
56  }
57  else
58  {
59  HashSet<FunctionDefinition> functions = context.GetFuncDefintionsInContract(baseContract);
60  foreach (FunctionDefinition function in functions)
61  {
62  if (function.Visibility == EnumVisibility.PRIVATE) continue;
63 
64  string signature = TransUtils.ComputeFunctionSignature(function);
65  context.AddFunctionToDynamicType(signature, contract, function);
66  }
67  // Events
68  // TODO: Do we need to lookup by signature?
69  HashSet<EventDefinition> events = context.GetEventDefintionsInContract(baseContract);
70  foreach (var evt in events)
71  {
72  context.AddEventToContract(contract, evt);
73  }
74  }
75  }
76  }
