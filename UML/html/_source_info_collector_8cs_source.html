76  private List<int> computeLineBreaks(string filePath)
77  {
78  List<int> LineBreaks = new List<int>();
79  int CharCount = 0;
80  LineBreaks.Add(0);
81  StreamReader file = new StreamReader(filePath);
82  string src = file.ReadToEnd();
83  file.Close();
84 
85  // About newline chars in UNIX and Windows:
86  // The only valid new chars are \r\n and \n, but not \r or \n\r
87  // The following code does make sure that if there are \r or \n\r in the file, it doesn't cause infinite looping.
88  // However, the code is either not properly translated or ConcurrencyExplorer shows highlight multiple lines at each step.
89  // The valid chars \r\n and \n can be mixed in a sol file.
90  int pos_r, pos_n;
91  do
92  {
93  pos_r = src.IndexOf('\r');
94  pos_n = src.IndexOf('\n');
95  if (pos_r >= 0 && (pos_r < pos_n || pos_n < 0))
96  {
97  CharCount += pos_r + 1;
98  LineBreaks.Add(CharCount);
99  if (pos_r + 1 == pos_n)
100  {
101  CharCount++;
102  pos_r++;
103  }
104  src = src.Substring(pos_r + 1);
105  }
106  else if (pos_n >= 0 && (pos_n < pos_r || pos_r < 0))
107  {
108  CharCount += pos_n + 1;
109  LineBreaks.Add(CharCount);
110  if (pos_n + 1 == pos_r)
111  {
112  CharCount++;
113  pos_n++;
114  }
115  src = src.Substring(pos_n + 1);
116  }
117  } while (pos_r >= 0 || pos_n >= 0);
118  LineBreaks.Add(CharCount + src.Length);
119  return LineBreaks;
120  }
121  }
