70  // Returns uintXX type of an unsigned integer constant, where XX is 8, 16, 24, ..., 256
71  private static int GetConstantSize(BigInteger constant)
72  {
73  BigInteger power = 1;
74  int size = 0;
75  while (power <= constant)
76  {
77  //power *= 2;
78  power <<= 1;
79  size++;
80  }
81 
82  if (power == constant) size--;
83 
84  if (size <= 8) return 8;
85  else if (size > 256) return -1;
86  else if (size % 8 == 0) return size;
87  else return 8 * (size / 8) + 8;
88  }

124  // "Out" parameter "value" is only used when IsUintConst is called in the context off a BinaryOperation on constants with the power operator.
125  public static bool IsUintConst(this TypeDescription typeDescription, Expression constant, out BigInteger value, out uint sz)
126  {
127  sz = 256;
128  try
129  {
130  var type = typeDescription.ToString();
131  if (type.StartsWith("int_const"))
132  {
133  if (constant is Literal)
134  {
135  string constStr = ((Literal)constant).Value;
136  value = BigInteger.Parse(constStr);
137  int res = GetConstantSize(value);
138  if (res == -1)
139  {
140  // Constant > 2^256:
141  Console.WriteLine($"Warning: uint constant > 2**256");
142  sz = 256;
143  return true;
144  }
145  else
146  {
147  sz = (uint)res;
148  return true;
149  }
150  }
151  else if (constant is BinaryOperation)
152  {
153  BinaryOperation binaryConstantExpr = (BinaryOperation)constant;
154 
155  if (binaryConstantExpr.LeftExpression is Literal && binaryConstantExpr.RightExpression is Literal)
156  {
157  // Get expression value from TypeDescriptions (but not if the constant is too big)
158  string typeStr = binaryConstantExpr.TypeDescriptions.TypeString;
159 
160  if (typeDescription.TypeString.Contains("omitted"))
161  {
162  // Large constant which is abbreviated in the typeDescription:
163  // This would probably never happen, because Solidity compiler reports type errors for such constants
164  Console.WriteLine($"Warning: constant expression contains large constant(s);");
165  Console.WriteLine($"if the large constant is an operand in a power operation,");
166  Console.WriteLine($"modulo arithmetic check will not work correctly");
167  value = 0;
168  sz = 256;
169  return false;
170  }
171  else
172  {
173  value = BigInteger.Parse(GetNumberFromEnd(typeStr));
174  sz = 256;
175  return true;
176  }
177  }
178  else
179  {
180  // TODO: The warning below could probably be suppressed:
181  // this is the case of exprs like (2+2+2...), where left and right expressions are BinaryOperation;
182  // in these cases, "value" is never used (it is only used for power operation on two literals)
183  Console.WriteLine($"Warning in IsUintConst: constant expression is neither Literal nor BinaryOperation with Literal operands; hint: use temps for subexpressions");
184  value = 0;
185  sz = 256;
186  return false;
187  }
188  }
189  else
190  {
191  // TODO: constant is TupleExpression:
192  Console.WriteLine($"Warning in IsUintConst: constant expression is neither Literal nor BinaryOperation; hint: use temps for subexpressions");
193  value = 0;
194  sz = 256;
195  return false;
196  }
197  }
198  else
199  {
200  sz = 256;
201  value = 0;
202  return false;
203  }
204  }
205  catch (Exception e)
206  {
207  Console.WriteLine($"VeriSol translation error: exception in IsUintConst: {e.Message}");
208  value = 0;
209  sz = 256;
210  return false;
211  }
212  }

570  private static string InferTypeFromTypeString(string typestr)
571  {
572  string typeString = typestr;
573  if (typeString.StartsWith("int_const"))
574  {
575  typeString = "int256";
576  }

578  {

583  typeString = "string";
584  }

587  typeString = "bytes"; //"bytes storage ref"
588  }
589  if (typeString.Contains(" memory")) //"struct Foo memory"
590  {
591  typeString = typeString.Substring(0, typeString.IndexOf(" memory"));
592  }
593  if (typeString.Contains(" storage"))
594  {
595  typeString = typeString.Substring(0, typeString.IndexOf(" storage"));
596  }
597  if (typeString.Contains(" payable"))
598  {
599  typeString = typeString.Substring(0, typeString.IndexOf(" payable")); //address payable
600  }
601  if (typeString.Contains(" calldata")) //"address[] calldata"
602  {
603  typeString = typeString.Substring(0, typeString.IndexOf(" calldata"));
604  }
605 
606 
607  return typeString;

710  public static List<BoogieVariable> CollectLocalVars(List<ContractDefinition> contracts, TranslatorContext context)
711  {
712  Debug.Assert(contracts.Count > 0, "Internal exception: expecting at least one contract here");
713  List<BoogieVariable> localVars = new List<BoogieVariable>()
714  {
715  new BoogieLocalVariable(new BoogieTypedIdent("this", BoogieType.Ref)),
716  new BoogieLocalVariable(new BoogieTypedIdent("msgsender_MSG", BoogieType.Ref)),
717  new BoogieLocalVariable(new BoogieTypedIdent("msgvalue_MSG", BoogieType.Int)),
718  new BoogieLocalVariable(new BoogieTypedIdent("choice", BoogieType.Int)),
719  };
720 
721  // use to remove duplicated variables by name
722  HashSet<string> uniqueVarNames = new HashSet<string>() { "this", "msgsender_MSG", "msgvalue_MSG", "choice" };
723 
724  foreach (var contract in contracts)
725  {
726  if (contract.ContractKind != EnumContractKind.CONTRACT) continue;
727 
728  // Consider all visible functions
729  HashSet<FunctionDefinition> funcDefs = context.GetVisibleFunctionsByContract(contract);
730  foreach (FunctionDefinition funcDef in funcDefs)
731  {
732  if (funcDef.Visibility == EnumVisibility.PUBLIC || funcDef.Visibility == EnumVisibility.EXTERNAL)
733  {
734  var inpParamCount = 0;
735  foreach (VariableDeclaration param in funcDef.Parameters.Parameters)
736  {
737  string name = $"__arg_{inpParamCount++}_" + funcDef.Name;
738  if (!string.IsNullOrEmpty(param.Name))
739  {
740  name = TransUtils.GetCanonicalLocalVariableName(param, context);
741  }
742  if (!uniqueVarNames.Contains(name))
743  {
744  BoogieType type = TransUtils.GetBoogieTypeFromSolidityTypeName(param.TypeName);
745  BoogieVariable localVar = new BoogieLocalVariable(new BoogieTypedIdent(name, type));
746  localVars.Add(localVar);
747  uniqueVarNames.Add(name);
748  }
749  }

773  /// <summary>
774  /// generate a non-deterministic choice block to call every public visible functions except constructors
775  ///
776  /// </summary>
777  /// <param name="contracts"></param>
778  /// <param name="context"></param>
779  /// <param name="callBackTarget">If non-null, this will be the msg.sender for calling back into the contracts</param>
780  /// <returns></returns>
781  public static BoogieIfCmd GenerateChoiceBlock(List<ContractDefinition> contracts, TranslatorContext context, Tuple<string, string> callBackTarget = null)
782  {
783  BoogieIfCmd ifCmd = null;
784  int j = 0;
785  foreach (var contract in contracts)
786  {
787  if (contract.ContractKind != EnumContractKind.CONTRACT) continue;
788 
789  HashSet<FunctionDefinition> funcDefs = context.GetVisibleFunctionsByContract(contract);
790  List<FunctionDefinition> publicFuncDefs = new List<FunctionDefinition>();
791  foreach (FunctionDefinition funcDef in funcDefs)
792  {
793  if (funcDef.IsConstructor) continue;
794  if (funcDef.IsFallback) continue; //let us not call fallback directly in harness
795  if (funcDef.Visibility == EnumVisibility.PUBLIC || funcDef.Visibility == EnumVisibility.EXTERNAL)
796  {
797  publicFuncDefs.Add(funcDef);
798  }
799  }
800  for (int i = publicFuncDefs.Count - 1; i >= 0; --i)
801  {
802  j++;
803  BoogieExpr guard = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ,
804  new BoogieIdentifierExpr("choice"),
805  new BoogieLiteralExpr(j));
806 
807  BoogieStmtList thenBody = new BoogieStmtList();
808 
809  FunctionDefinition funcDef = publicFuncDefs[i];
810  string callee = TransUtils.GetCanonicalFunctionName(funcDef, context);
811  List<BoogieExpr> inputs = new List<BoogieExpr>()
812  {
813  // let us just call back into the calling contract
814  callBackTarget != null ? new BoogieIdentifierExpr(callBackTarget.Item1) : new BoogieIdentifierExpr("this"),
815  callBackTarget != null ? new BoogieIdentifierExpr(callBackTarget.Item2) : new BoogieIdentifierExpr("msgsender_MSG"),
816  new BoogieIdentifierExpr("msgvalue_MSG"),
817  };
818  var inpParamCount = 0;
819  foreach (VariableDeclaration param in funcDef.Parameters.Parameters)
820  {
821  string name = $"__arg_{inpParamCount++}_" + funcDef.Name;
822  if (!string.IsNullOrEmpty(param.Name))
823  {
824  name = TransUtils.GetCanonicalLocalVariableName(param, context);
825  }
826  inputs.Add(new BoogieIdentifierExpr(name));
827  if (param.TypeName is ArrayTypeName array)
828  {
829  thenBody.AddStatement(new BoogieCallCmd(
830  "FreshRefGenerator",
831  new List<BoogieExpr>(), new List<BoogieIdentifierExpr>() { new BoogieIdentifierExpr(name) }));
832  }
833 
834  }
835 
836  List<BoogieIdentifierExpr> outputs = new List<BoogieIdentifierExpr>();
837  var retParamCount = 0;
838 
839  foreach (VariableDeclaration param in funcDef.ReturnParameters.Parameters)
840  {
841  string name = $"__ret_{retParamCount++}_" + funcDef.Name;
842  if (!string.IsNullOrEmpty(param.Name))
843  {
844  name = TransUtils.GetCanonicalLocalVariableName(param, context);
845 
846  }
847 
848  outputs.Add(new BoogieIdentifierExpr(name));
849  }
850 
851  if (context.TranslateFlags.InstrumentGas)
852  {
853  var gasVar = new BoogieIdentifierExpr("gas");
854  thenBody.AddStatement(new BoogieAssignCmd(gasVar, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.SUB, gasVar, new BoogieLiteralExpr(TranslatorContext.TX_GAS_COST))));
855  }
856  BoogieCallCmd callCmd = new BoogieCallCmd(callee, inputs, outputs);
857  thenBody.AddStatement(callCmd);
858 
859  BoogieStmtList elseBody = ifCmd == null ? null : BoogieStmtList.MakeSingletonStmtList(ifCmd);
860  ifCmd = new BoogieIfCmd(guard, thenBody, elseBody);
861  }
862  }
863  return ifCmd;
864  }
