211  private string[] FilterCorralTrace(string[] trace)
212  {
213  // Only get lines that contain "Trace: Thread=1":
214  var res = trace.Where(s => s.Contains("Trace: Thread=1"));
215  // Remove irrelevant lines: "()", "(Done)", "(x = 0)":
216  res = res.Where(x => x.Contains("CALL ") || x.Contains("RETURN ") || x.Contains("ASSERTION FAILS ") || x.Contains("_verisolFirstArg"));
217  return res.ToArray();

220  private List<Tuple<string, string>> ConvertTrace(string[] corralTrace)
221  {
222  var res = new List<Tuple<string, string>>();
223  foreach (string line in corralTrace)
224  {
225  var strSplit = line.Split("Trace: Thread=1 ");
226  // This should never happen; TODO: Debug.Assert(false)?
227  if (strSplit.Count() == 0) continue;
228  // Strip braces from the 2nd string:
229  if (strSplit[1].Length > 2)
230  {
231  if (strSplit[1].StartsWith("(ASSERTION FAILS"))
232  {
233  // To avoid splitting the assert (which could contain "=="
234  // in the "ASSERT FAILS" line;
235  strSplit[1] = strSplit[1].Replace(", ", ",");
236  }
237  else if (strSplit[1].StartsWith("(") && strSplit[1].EndsWith(")"))
238  {
239  strSplit[1] = strSplit[1].Substring(1, strSplit[1].Length - 2);
240  }
241  }
242  var tuples = strSplit[1].Split(", ").ToList();
243  tuples.ForEach(x => res.Add(Tuple.Create(strSplit[0], x)));
244  }
245  return res;
246  }

248  private string ConvertFunctionName(string origName)
249  {
250  string[] nameSplit = origName.Split("_");
251  if (nameSplit.Count() > 2)
252  {
253  // More than one underscore in the name: leave it as it is:
254  //Console.WriteLine($"Name {origName} has more than one underscore and will not be converted into a user-friendly format");
255  return origName;
256  }
257  else if (nameSplit.Count() == 2)
258  {
259  if (nameSplit[0] == nameSplit[1])
260  {
261  // Top level constructor name "XX_XX":
262  return nameSplit[0] + "::Constructor";
263  }
264  else
265  {
266  // Regular function call: functionName_ConstractName
267  return nameSplit[1] + "::" + nameSplit[0];
268  }
269  }
270  else
271  {
272  Debug.Assert(false, $"Unreachable: Function name {origName} does not contain underscores");
273  return String.Empty;
274  }

287  private void PrintCounterexampleHelper(List<Tuple<string, string>> trace)
288  {
289  Stack<string> callStack = new Stack<string>();
290  string currentArgs = "";
291  bool collectArgs = false;
292  List<string> resultArray = new List<string>();
293 
294  // this is a list of (line#, element)
295  // element \in {CALL foo, RETURN from foo, x = e, ASSERTION FAILS, ...}
296  for (int i = 0; i < trace.Count(); i++)
297  {
298  var elem = trace[i].Item2;
299  if (elem.StartsWith("CALL CorralChoice_"))
300  {
301  continue;
302  }
303  else if (elem.StartsWith("RETURN from CorralChoice_"))
304  {
305  continue;
306  }
307  else if (elem.StartsWith("CALL "))
308  {
309  var func = elem.Substring("CALL ".Length);
310  callStack.Push(func);
311  currentArgs = "";
312  }
313  else if (elem.StartsWith("RETURN from "))
314  {
315  var func = elem.Substring("RETURN from ".Length);
316  Debug.Assert(callStack.Count > 0, "Call stack cannot be empty");
317  Debug.Assert(func.TrimEnd().Equals(callStack.Peek().TrimEnd()), $"Top of stack {callStack.Peek()} does not match with return {func}");
318  callStack.Pop();
319  }
320  else if (elem.StartsWith("_verisolFirstArg"))
321  {
322  collectArgs = true;
323  }
324  else if (elem.StartsWith("_verisolLastArg"))
325  {
326  Debug.Assert(callStack.Count > 0, "callstack cannot be empty");
327  collectArgs = false;
328  if (callStack.Count == 1)
329  {
330  resultArray.Add($"{trace[i].Item1}: {ConvertFunctionName(callStack.Peek())} ({currentArgs.Substring(", ".Length)})");
331  }
332  }
333  else if (elem.StartsWith("(ASSERTION FAILS"))
334  {
335  resultArray.Add($"{trace[i].Item1}: ASSERTION FAILS!");
336  }
337  else if (elem.Contains("=") && !elem.Contains("=="))
338  {
339 
340  if (collectArgs)
341  {
342  // Replace "T@Ref!val!0" with "address!0"
343  string[] splitElem = elem.Split(" = ");
344  Debug.Assert(splitElem.Count() == 2);
345  string[] rhsSplit = splitElem[1].Split("!");
346  if (rhsSplit.Count() > 0 && rhsSplit[0].Equals("T@Ref"))
347  {
348  currentArgs += ", " + elem.Replace("T@Ref!val", "address");
349  }
350  else
351  {
352  currentArgs += ", " + elem;
353  }
354  }
355  }
356  else
357  {
358  Debug.Assert(false, $"This should be unreachable, found a new class of statement {elem}");
359  }
360  }
361 
362  resultArray.ForEach(x => Console.WriteLine(x));
363  // Only printing counterexample on the command line:
364  File.WriteAllLines(counterexampleFileName, resultArray);
365  return;
366  }

368  private bool ExecuteSolToBoogie()
369  {

373  SolidityCompiler compiler = new SolidityCompiler();

389  Console.WriteLine($"... running SolToBoogie to translate Solidity to Boogie");

391 
392  // dump the Boogie program to a file

402  return false;
403  }
