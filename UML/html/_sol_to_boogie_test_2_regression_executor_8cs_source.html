52  public int BatchExecute()
53  {
54  string[] filePaths = Directory.GetFiles(testDirectory);
55  int passedCount = 0;
56  int failedCount = 0;
57  ReadRecord();
58  foreach (string filePath in filePaths)
59  {
60  string filename = Path.GetFileName(filePath);
61  if (!filename.StartsWith(testPrefix))
62  continue;
63  //silently ignore non .sol files
64  if (!filename.EndsWith(".sol"))
65  continue;
66  if (!filesToRun.ContainsKey(filename))
67  {
68  logger.LogWarning($"{filename} not found in {Path.Combine(recordsDir, "records.txt")}");
69  continue;
70  }
71 
72  if (!filesToRun[filename])
73  {
74  continue;
75  }
76 
77  logger.LogInformation($"Running {filename}");
78 
79  BatchExeResult batchExeResult = BatchExeResult.SolcError;
80  string expectedCorralOutput = "", currentCorralOutput = "";
81  try
82  {
83  batchExeResult = Execute(filename, out expectedCorralOutput, out currentCorralOutput);
84  }
85  catch (Exception exception)
86  {
87  logger.LogCritical(exception, $"Exception occurred in {filename}");
88  batchExeResult = BatchExeResult.OtherException;
89  }
90 
91  if (batchExeResult == BatchExeResult.Success)
92  {
93  ++passedCount;
94  logger.LogInformation($"Passed - {filename}");
95  }
96  else if (batchExeResult == BatchExeResult.SolcError)
97  {
98  ++failedCount;
99  logger.LogError($"Failed (Solc failed) - {filename}");
100  }
101  else if (batchExeResult == BatchExeResult.OtherException)
102  {
103  ++failedCount;
104  logger.LogError($"Failed (Other exception) - {filename}");
105  }
106  else if (batchExeResult == BatchExeResult.SolToBoogieError)
107  {
108  ++failedCount;
109  logger.LogError($"Failed (VeriSol translation error) - {filename}");
110  }
111  else if (batchExeResult == BatchExeResult.CorralError)
112  {
113  ++failedCount;
114  logger.LogError($"Failed (Corral regression failed) - {filename}");
115  logger.LogError($"\t Expected - {expectedCorralOutput}");
116  logger.LogError($"\t Corral detailed Output - {currentCorralOutput}");
117  }
118  else
119  {
120  ++failedCount;
121  logger.LogError($"Failed (Tool error: unexpected failure code) - {filename}");
122  }
123  }
124 
125  logger.LogInformation($"{passedCount} passed {failedCount} failed");
126  // To allow time for logging the last summary line:
127  System.Threading.Thread.Sleep(100);
128  DeleteTemporaryFiles();
129  return (failedCount == 0)? 0 : 1;
130  }

155  public BatchExeResult Execute(string filename, out string expected, out string current)
156  {
157  BatchExeResult result = BatchExeResult.SolcError;
158  string filePath = Path.Combine(testDirectory, filename);
159 
160  // compile the program
161  SolidityCompiler compiler = new SolidityCompiler();
162  CompilerOutput compilerOutput = compiler.Compile(solcPath, filePath);
163 
164  if (compilerOutput.ContainsError())
165  {
166  compilerOutput.PrintErrorsToConsole();
167  throw new SystemException("Compilation Error");
168  }
169 
170  // build the Solidity AST from solc output
171  AST solidityAST = new AST(compilerOutput, Path.GetDirectoryName(filePath));
172 
173  // read the corral configuration from Json
174  string configJsonPath = Path.Combine(configDirectory, Path.GetFileNameWithoutExtension(filename) + ".json");
175  string jsonString = File.ReadAllText(configJsonPath);
176  CorralConfiguration corralConfig = JsonConvert.DeserializeObject<CorralConfiguration>(jsonString);
177 
178  // translate Solidity to Boogie
179  try
180  {
181  BoogieTranslator translator = new BoogieTranslator();
182  var translatorFlags = new TranslatorFlags();
183  translatorFlags.GenerateInlineAttributes = false;
184  if (corralConfig.TranslatorOptions != null)
185  {
186  ParseTranslatorFlags(translatorFlags, corralConfig.TranslatorOptions);
187  }
188 
189  BoogieAST boogieAST = translator.Translate(solidityAST, new HashSet<Tuple<string, string>>(), translatorFlags);
190 
191  // dump the Boogie program to a file
192  using (var outWriter = new StreamWriter(outFile))
193  {
194  outWriter.WriteLine(boogieAST.GetRoot());
195  }
196  } catch (Exception e)
197  {
198  Console.WriteLine($"VeriSol translation error: {e.Message}");
199  result = BatchExeResult.SolToBoogieError;
200  expected = current = null;
201  return result;
202  }
203 
204  // read the corral configuration from Json
205  //string configJsonPath = Path.Combine(configDirectory, Path.GetFileNameWithoutExtension(filename) + ".json");
206  //string jsonString = File.ReadAllText(configJsonPath);
207  //CorralConfiguration corralConfig = JsonConvert.DeserializeObject<CorralConfiguration>(jsonString);
208 
209  string corralOutput = RunCorral(corralConfig);
210  expected = corralConfig.ExpectedResult;
211  current = corralOutput;
212  result = CompareCorralOutput(corralConfig.ExpectedResult, corralOutput);
213  return result;
214  }
