25  public void Generate()
26  {
27  List<BoogieVariable> inParams = new List<BoogieVariable>()
28  {
29  new BoogieFormalParam(new BoogieTypedIdent("from", BoogieType.Ref)),
30  new BoogieFormalParam(new BoogieTypedIdent("to", BoogieType.Ref)),
31  new BoogieFormalParam(new BoogieTypedIdent("amount", BoogieType.Int))
32  };
33  List<BoogieVariable> outParams = new List<BoogieVariable>();
34 
35  List<BoogieAttribute> attributes = new List<BoogieAttribute>();
36  if (context.TranslateFlags.GenerateInlineAttributes)
37  {
38  attributes.Add(new BoogieAttribute("inline", 1));
39  };
40  var procName = "FallbackDispatch";
41  BoogieProcedure procedure = new BoogieProcedure(procName, inParams, outParams, attributes);
42  context.Program.AddDeclaration(procedure);
43 
44  List<BoogieVariable> localVars = new List<BoogieVariable>();
45  var fbUnknownProcName = "Fallback_UnknownType";
46  BoogieStmtList procBody = GenerateBodyOfFallbackDispatch(inParams, fbUnknownProcName);
47 
48  BoogieImplementation implementation = new BoogieImplementation(procName, inParams, outParams, localVars, procBody);
49  context.Program.AddDeclaration(implementation);
50 
51  // let us havoc all the global variables when fallback_unknowntype is called
52  var modSet = context.Program.Declarations.Where(x => x is BoogieGlobalVariable).Select(x => (BoogieGlobalVariable)x).ToList();
53  BoogieProcedure unknownFbProc = new BoogieProcedure(fbUnknownProcName, inParams, outParams, attributes);
54  context.Program.AddDeclaration(unknownFbProc);
55 
56  // we need to create an implementation as Corral seem to ignore modifies on declarations
57  // https://github.com/boogie-org/corral/issues/98
58  BoogieStmtList fbBody = new BoogieStmtList();
59  var fbLocalVars = new List<BoogieVariable>();
60 
61  if (context.TranslateFlags.ModelStubsAsSkips() || context.TranslateFlags.ModelStubsAsCallbacks())
62  {
63  fbBody.AppendStmtList(CreateBodyOfUnknownFallback(fbLocalVars, inParams));
64  }
65  else
66  {
67  Debug.Assert(context.TranslateFlags.ModelStubsAsHavocs(), "Unknown option for modeling stubs");
68  foreach (var global in modSet)
69  {
70  fbBody.AddStatement(new BoogieHavocCmd(new BoogieIdentifierExpr(global.Name)));
71  }
72  }
73 
74  BoogieImplementation unknownFbImpl = new BoogieImplementation(fbUnknownProcName, inParams, outParams, fbLocalVars, fbBody);
75  context.Program.AddDeclaration(unknownFbImpl);
76 
77  var sendProcName = "send";
78 
79  var sendOutParams = new List<BoogieVariable>(){ new BoogieFormalParam(new BoogieTypedIdent("success", BoogieType.Bool))};
80  BoogieProcedure sendProc = new BoogieProcedure(sendProcName, inParams, sendOutParams, attributes);
81 
82  context.Program.AddDeclaration(sendProc);
83 
84  BoogieStmtList sendBody = CreateBodyOfSend(inParams, sendOutParams, procName);
85  context.Program.AddDeclaration(new BoogieImplementation(sendProcName, inParams, sendOutParams, localVars, sendBody));
86  }

88  private BoogieStmtList CreateBodyOfSend(List<BoogieVariable> inParams, List<BoogieVariable> outParams, string fbProcName)
89  {
90  var fromIdExp = new BoogieIdentifierExpr(inParams[0].Name);
91  var amtIdExp = new BoogieIdentifierExpr(inParams[2].Name);
92  var guard = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.GE,
93  new BoogieMapSelect(new BoogieIdentifierExpr("Balance"), fromIdExp),
94  amtIdExp);
95 
96  // call FallbackDispatch(from, to, amount)
97  var toIdExpr = new BoogieIdentifierExpr(inParams[1].Name);
98  var callStmt = new BoogieCallCmd(
99  fbProcName,
100  new List<BoogieExpr>() { fromIdExp, toIdExpr, amtIdExp},
101  new List<BoogieIdentifierExpr>()
102  );
103 
104  var thenBody = new BoogieStmtList();
105  var successIdExp = new BoogieIdentifierExpr(outParams[0].Name);
106  thenBody.AddStatement(callStmt);
107  thenBody.AddStatement(new BoogieAssignCmd(successIdExp, new BoogieLiteralExpr(true)));
108 
109  var elseBody = new BoogieAssignCmd(successIdExp, new BoogieLiteralExpr(false));
110 
111  return BoogieStmtList.MakeSingletonStmtList(new BoogieIfCmd(guard, thenBody,
112  BoogieStmtList.MakeSingletonStmtList(elseBody)));
113  }

115  private BoogieStmtList CreateBodyOfUnknownFallback(List<BoogieVariable> fbLocalVars, List<BoogieVariable> inParams)
116  {
117 
118  Debug.Assert(context.TranslateFlags.ModelStubsAsSkips() || context.TranslateFlags.ModelStubsAsCallbacks(),
119  "CreateBodyOfUnknownFallback called in unexpected context");
120  var procBody = new BoogieStmtList();
121 
122  procBody.AddStatement(new BoogieCommentCmd("---- Logic for payable function START "));
123  var balnSender = new BoogieMapSelect(new BoogieIdentifierExpr("Balance"), new BoogieIdentifierExpr(inParams[0].Name));
124  var balnThis = new BoogieMapSelect(new BoogieIdentifierExpr("Balance"), new BoogieIdentifierExpr(inParams[1].Name));
125  var msgVal = new BoogieIdentifierExpr("amount");
126  //assume Balance[msg.sender] >= msg.value
127  procBody.AddStatement(new BoogieAssumeCmd(new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.GE, balnSender, msgVal)));
128  //balance[from] = balance[from] - msg.value
129  procBody.AddStatement(new BoogieAssignCmd(balnSender, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.SUB, balnSender, msgVal)));
130  //balance[to] = balance[to] + msg.value
131  procBody.AddStatement(new BoogieAssignCmd(balnThis, new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.ADD, balnThis, msgVal)));
132  procBody.AddStatement(new BoogieCommentCmd("---- Logic for payable function END "));
133 
134  if (context.TranslateFlags.ModelStubsAsCallbacks())
135  {
136  fbLocalVars.AddRange(TransUtils.CollectLocalVars(context.ContractDefinitions.ToList(), context));
137  // if (*) fn1(from, *, ...)
138  // we only redirect the calling contract, but the msg.sender need not be to, as it can call into anohter contract that calls
139  // into from
140  procBody.AddStatement(TransUtils.GenerateChoiceBlock(context.ContractDefinitions.ToList(), context, Tuple.Create(inParams[0].Name, inParams[1].Name)));
141  }
142  return procBody;
143  }

145  private BoogieStmtList GenerateBodyOfFallbackDispatch(List<BoogieVariable> inParams, string fbUnknownProcName)
146  {
147  // Perform the payable logic to transfer balance
148  //
149  // Fallback(from, to, amount)
150  //
151  // foreach contract C that is not Lib/VeriSol
152  // if (DT[this] == C)
153  // if C has a fallback f
154  // call ret := fallBack_C(this=to, sender=from, msg.value)
155  // else
156  // assume msg.value == 0;
157  // else
158  // if stubModel == callback
159  // call fallBack_unknownType(from, to, amout) //we switch the order of first two parameters to ease callbacks
160  // else if stubmodel == havoc
161  // havoc all global state, except local state of this contract
162  // else
163  // skip //default
164  BoogieIfCmd ifCmd = null;
165 
166  Debug.Assert(context.ContractDefinitions.Count >= 1, "There should be at least one contract");
167 
168  List<BoogieIdentifierExpr> outParams = new List<BoogieIdentifierExpr>();
169 
170  List<BoogieExpr> unkwnFnArgs = new List<BoogieExpr>()
171  {
172  new BoogieIdentifierExpr(inParams[0].Name),
173  new BoogieIdentifierExpr(inParams[1].Name),
174  new BoogieIdentifierExpr(inParams[2].Name)
175  };
176 
177  // fbUnknown(from, to, amount)
178  BoogieStmtList noMatchCase = BoogieStmtList.MakeSingletonStmtList(
179  new BoogieCallCmd(fbUnknownProcName, unkwnFnArgs, outParams));
180 
181  foreach (var contract in context.ContractDefinitions)
182  {
183  if (contract.ContractKind == EnumContractKind.LIBRARY) continue;
184 
185  FunctionDefinition function = context.ContractToFallbackMap.ContainsKey(contract) ?
186  context.ContractToFallbackMap[contract] : null;
187  BoogieExpr lhs = new BoogieMapSelect(new BoogieIdentifierExpr("DType"), new BoogieIdentifierExpr(inParams[1].Name));
188  BoogieExpr rhs = new BoogieIdentifierExpr(contract.Name);
189  BoogieExpr guard = new BoogieBinaryOperation(BoogieBinaryOperation.Opcode.EQ, lhs, rhs);
190 
191 
192  BoogieStmtList thenBody = null;
193 
194  if (function != null)
195  {
196  List<BoogieExpr> arguments = new List<BoogieExpr>()
197  {
198  new BoogieIdentifierExpr(inParams[1].Name),
199  new BoogieIdentifierExpr(inParams[0].Name),
200  new BoogieIdentifierExpr(inParams[2].Name)
201  };
202 
203  string callee = TransUtils.GetCanonicalFunctionName(function, context);
204  // to.fallback(from, amount), thus need to switch param0 and param1
205  thenBody = BoogieStmtList.MakeSingletonStmtList(new BoogieCallCmd(callee, arguments, outParams));
206  }
207  else
208  {
209  // No fallback means not payable (amount == 0)
210  thenBody = BoogieStmtList.MakeSingletonStmtList(
211  new BoogieAssumeCmd(
212  new BoogieBinaryOperation(
213  BoogieBinaryOperation.Opcode.EQ,
214  new BoogieIdentifierExpr(inParams[2].Name),
215  new BoogieLiteralExpr(BigInteger.Zero)))
216  );
217  }
218 
219  BoogieStmtList elseBody = ifCmd == null ? noMatchCase : BoogieStmtList.MakeSingletonStmtList(ifCmd);
220 
221  ifCmd = new BoogieIfCmd(guard, thenBody, elseBody);
222  }
223 
224  return BoogieStmtList.MakeSingletonStmtList(ifCmd);
225  }
226  }
